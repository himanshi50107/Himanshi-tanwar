  const canvas = document.getElementById("dnaCanvas");
  const ctx = canvas.getContext("2d");

  let angle = 0;

  function drawDNA() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);

    for (let i = 0; i < 20; i++) {
      let x = 200 + Math.sin(i * 0.5 + angle) * 80;
      let y = i * 15;

      let x2 = 200 + Math.sin(i * 0.5 + angle + Math.PI) * 80;

      ctx.beginPath();
      ctx.arc(x, y, 5, 0, Math.PI * 2);
      ctx.fill();

      ctx.beginPath();
      ctx.arc(x2, y, 5, 0, Math.PI * 2);
      ctx.fill();

      ctx.beginPath();
      ctx.moveTo(x, y);
      ctx.lineTo(x2, y);
      ctx.stroke();
    }

    angle += 0.05;
    requestAnimationFrame(drawDNA);
  }

  drawDNA();
